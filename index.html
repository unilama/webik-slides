Ciekawostka

function fib_iter($n)
{
    if($n <= 2){
        return 1;
    }else{
        $a = 1;
        $b = 1;
        $c = 0;
        for($i=0; $i<$n-2; $i++){
            $c = $a + $b;
            $a = $b;
            $b = $c;
        }
        return $c;
    }
}

function fib_rek($n)
{
	if($n==0)
	{
		    return 0;
	}
	if($n==1)
	{
		    return 1;
	}
	return fib_rek($n-2) + fib_rek($n-1);
}
